SQL Setup — Supabase
Ejecuta esto en Supabase → SQL Editor para crear todas las tablas:
create table if not exists metrics ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, created_at timestamptz default now(), name text not null, spend numeric default 0, roas numeric default 0, purchases int default 0, cpa numeric default 0, ctr numeric default 0, cpm numeric default 0, cpc numeric default 0, frequency numeric default 0, hook_rate numeric default 0, p25 numeric default 0, p75 numeric default 0, p100 numeric default 0, notes text default '', snapshot_date date default current_date ); alter table metrics enable row level security; create policy "own" on metrics for all using (auth.uid()=user_id); -- Columnas adicionales para snapshots Meta (ejecutar si ya tenés la tabla) alter table metrics add column if not exists snap_type text default 'manual'; alter table metrics add column if not exists meta_id text; alter table metrics add column if not exists snap_time timestamptz; create table if not exists scripts ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, created_at timestamptz default now(), updated_at timestamptz default now(), title text default 'Nuevo guion', angle text default '', product text default '', hook text default '', body text default '', cta text default '', copy_feed text default '', hook_rate numeric default 0, p25 numeric default 0, p75 numeric default 0, p100 numeric default 0, status text default 'borrador', notes text default '' ); alter table scripts enable row level security; create policy "own" on scripts for all using (auth.uid()=user_id); create table if not exists objectives ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, created_at timestamptz default now(), name text not null, target numeric default 0, current_val numeric default 0, unit text default '', month text default '' ); alter table objectives enable row level security; create policy "own" on objectives for all using (auth.uid()=user_id); create table if not exists finances ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, created_at timestamptz default now(), product text not null, units int default 0, revenue numeric default 0, cost_prod numeric default 0, cost_ads numeric default 0, profit numeric default 0, margin numeric default 0, period text default '' ); alter table finances enable row level security; create policy "own" on finances for all using (auth.uid()=user_id); create table if not exists content_posts ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, created_at timestamptz default now(), date date default current_date, platform text default 'Instagram', format text default 'Reel', caption text default '', status text default 'pendiente', notes text default '' ); alter table content_posts enable row level security; create policy "own" on content_posts for all using (auth.uid()=user_id); create table if not exists contacts ( id uuid default gen_random_uuid() primary key, user_id uuid references auth.users not null, created_at timestamptz default now(), name text not null, brand text default '', service text default '', status text default 'lead', notes text default '' ); alter table contacts enable row level security; create policy "own" on contacts for all using (auth.uid()=user_id);
